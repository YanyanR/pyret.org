data BinTree:
  | leaf
  | node(value, left, right)
end

fun tree-sum(t):
  cases(BinTree) t:
    | leaf => 0
    | node(v, l, r) => v + tree-sum(l) + tree-sum(r)
  end
where:
  tree-sum(leaf) is 0
  tree-sum(node(5, node(4, leaf, leaf), leaf)) is 9
end

fun sum(l :: List):
  cases(List) l:
    | empty => 0
    | link(first, rest) => first + sum(rest)
  end
where:
  sum([]) is 0
  sum([1, 2, 3]) is 6
end
